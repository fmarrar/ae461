Question 1

Part a

vf = [0.6-0.05, 0.6+0.05];
vm = ones(1, length(vf)) - vf;

Em = 2.4e9;
Ef = 120e9;
Gm = Em/(2*(1 + 0.35));
Gf = Ef/(2*(1 + 0.32));

Ex = vf*Ef + vm*Em;
Ey = 1./(1/Ef * vf + 1/Em * vm);
vxy = vf.*vf + vm.*vm;
Gxy = 1./(1/Gf * vf + 1/Gm * vm);

% Part b
Exavg = mean(Ex);
Exstd = std(Ex);

Eyavg = mean(Ey);
Eystd = std(Ey);

vxyavg = mean(vxy);
vxystd = std(vxy);

Gxyavg = mean(Gxy);
Gxystd = std(Gxy);
